# choose the traning and test features 
X = pca_df
y = df['Outcome']

# set up the decision tree model 
dt_clf = DecisionTreeClassifier(random_state=2)

param_grid = {
    'criterion': ['gini', 'entropy'],
    'max_depth': [None, 2, 4, 6, 8, 10],
    'min_samples_split': [2, 5, 10],
    'min_samples_leaf': [1, 2, 4]
}

# do the cross validation and find the best parameter 
grid_search = GridSearchCV(dt_clf, param_grid, cv=5, scoring='accuracy')
grid_search.fit(X, y)

best_params = grid_search.best_params_
best_dtree = DecisionTreeClassifier(**best_params)
print("Best Parameters: ", grid_search.best_params_)

# perform k-folder cross-validation
accuracy_scores = []
precision_scores = []
recall_scores = []
f1_scores = []

k = 10
kf = KFold(n_splits=k)

# Initialize the total confusion matrix
total_conf_matrix = np.zeros((2,2)) 

for train_index, test_index in kf.split(X):
    X_train, X_test = X.iloc[train_index], X.iloc[test_index]
    y_train, y_test = y.iloc[train_index], y.iloc[test_index]

    best_dtree.fit(X_train, y_train)
    y_pred = best_dtree.predict(X_test)

    accuracy = accuracy_score(y_test, y_pred)
    precision = precision_score(y_test, y_pred)
    recall= recall_score(y_test, y_pred)
    f1 = f1_score(y_test, y_pred)
    
    accuracy_scores.append(accuracy)
    precision_scores.append(precision)
    recall_scores.append(recall)
    f1_scores.append(f1)
    conf_matrix = confusion_matrix(y_test, y_pred)
    
    # Add the new confusion matrix to the total confusion matrix
    total_conf_matrix += conf_matrix 

# print the mean cross-validation scores
print("Mean Cross-Validation Accuracy: ", np.mean(accuracy_scores))

from sklearn.model_selection import KFold
from sklearn.metrics import accuracy_score, confusion_matrix, precision_score, recall_score, f1_score
from sklearn.pipeline import Pipeline
from sklearn.preprocessing import StandardScaler
from sklearn.svm import SVC

# choose the training and test features
X = pca_df
y = df['Outcome']  

# set up the k-folder
k = 10
kf = KFold(n_splits=k)

# set up the SVM model with the best parameters
best_params = {'C': 10, 'kernel': 'linear', 'gamma': 'auto'}
svm = SVC(**best_params)

# create a pipeline with the scaler and the svm model
best_svm = Pipeline([
    ('scaler', StandardScaler()),
    ('svm', svm)
])

# perform k-folder cross-validation
accuracy_scores = []
precision_scores = []
recall_scores = []
f1_scores = []
total_conf_matrix_3 = np.zeros((2,2)) 
for train_index, test_index in kf.split(X):
    X_train, X_test = X.iloc[train_index], X.iloc[test_index]
    y_train, y_test = y.iloc[train_index], y.iloc[test_index]
    
    # train the SVM model
    best_svm.fit(X_train, y_train)
    
    # make predictions
    y_pred = best_svm.predict(X_test)
    conf_matrix_3 = confusion_matrix(y_test, y_pred)
    
    # Add the new confusion matrix to the total confusion matrix
    total_conf_matrix_3 += conf_matrix_3
    # get the accuracy, precision, recall, and F1 scores
    accuracy_svm = accuracy_score(y_test, y_pred)
    precision_svm = precision_score(y_test, y_pred)
    recall_svm = recall_score(y_test, y_pred)
    f1_svm = f1_score(y_test, y_pred)
    
    accuracy_scores.append(accuracy_svm)
    precision_scores.append(precision_svm)
    recall_scores.append(recall_svm)
    f1_scores.append(f1_svm)

# print the mean cross-validation scores
print("Mean Cross-Validation Accuracy: ", np.mean(accuracy_scores))
print("Mean Cross-Validation Precision: ", np.mean(precision_scores))
print("Mean Cross-Validation Recall: ", np.mean(recall_scores))
print("Mean Cross-Validation F1 Score: ", np.mean(f1_scores))
